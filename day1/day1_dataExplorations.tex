\begin{frame}[fragile]\frametitle{Calculations with assignment statements}
We can store a value(s) in an R object using the assignment symbol $<-$ ("less than" followed by a hyphen)
\begin{Schunk}
\begin{Sinput}
> x <- 2.5
\end{Sinput}
\end{Schunk}
To check what is in a variable type the variable name
\begin{Schunk}
\begin{Sinput}
> x
\end{Sinput}
\begin{Soutput}
[1] 2.5
\end{Soutput}
\end{Schunk}
Can store a computation under a new R object or change the current value stored in an old object
\begin{Schunk}
\begin{Sinput}
> y <- 3*log(x)
\end{Sinput}
\end{Schunk}
\end{frame}

\begin{frame}[fragile]\frametitle{Search and Replace}
\begin{Schunk}
\begin{Sinput}
> x <- "Lorem Ipsum is simply dummy text of the 
+ printing and typesetting industry. This is for the dummy text"
> grep("the", x)
\end{Sinput}
\begin{Soutput}
[1] 1
\end{Soutput}
\begin{Sinput}
> grep("the",x,value=TRUE)
\end{Sinput}
\begin{Soutput}
[1] "Lorem Ipsum is simply dummy text of the \nprinting and typesetting industry. This is for the dummy text"
\end{Soutput}
\begin{Sinput}
> #first replacement 
> x2 <- sub( "the" , "THE",x)
> x2
\end{Sinput}
\begin{Soutput}
[1] "Lorem Ipsum is simply dummy text of THE \nprinting and typesetting industry. This is for the dummy text"
\end{Soutput}
\begin{Sinput}
> #All replacement 
> x3 <- gsub( "the" , "THE",x)
> x3
\end{Sinput}
\begin{Soutput}
[1] "Lorem Ipsum is simply dummy text of THE \nprinting and typesetting industry. This is for THE dummy text"
\end{Soutput}
\end{Schunk}

\end{frame}
